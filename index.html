-- Objetivo: Identificar el modelo y MAC Address del equipo asignado al servicio del cliente.
SELECT
    S.TIPO_SERVICIO,
    E.MODELO,
    E.MAC_ADDRESS
FROM
    Servicios AS S
JOIN
    Equipos AS E ON S.ID_SERVICIO = E.ID_SERVICIO
WHERE
    S.ID_SUSCRIPTOR = 'CLXXXX'
    AND S.TIPO_SERVICIO = 'Internet'; -- Filtrar por el tipo de servicio que falla.
